SELECT AVG(P.SALARIO)
FROM UNIVERSIDADE.PROFESSOR P
WHERE P.CATEGORIA = 'Adjunto I';

--2)
SELECT COUNT(D)
FROM UNIVERSIDADE.DISCIPLINA D
WHERE D.CARGA_HORARIA = 60;

--3)
SELECT P.NOME, D.NOME
FROM UNIVERSIDADE.PROFESSOR P, UNIVERSIDADE.DISCIPLINA D, UNIVERSIDADE.MINISTRA M
WHERE P.SIAPE = M.SIAPE AND D.COD_DISCIPLINA = M.COD_DISCIPLINA AND M.PERIODO = '2013.2';

--4)
SELECT P.NOME
FROM UNIVERSIDADE.PROFESSOR P
WHERE  P.SIAPE NOT IN(
	SELECT DISTINCT M.SIAPE
	FROM UNIVERSIDADE.MINISTRA M);
	

--5)
SELECT P.NOME
FROM UNIVERSIDADE.PROFESSOR P
WHERE NOT EXISTS(
	SELECT M.SIAPE
	FROM UNIVERSIDADE.MINISTRA M
	WHERE M.PERIODO = '2013.1' AND P.SIAPE = M.SIAPE);

--6)
SELECT D.NOME
FROM UNIVERSIDADE.DISCIPLINA D
WHERE  (SELECT COUNT(M.*)
		FROM UNIVERSIDADE.MINISTRA M
		WHERE M.COD_DISCIPLINA = D.COD_DISCIPLINA) > 2;

--7)
SELECT COUNT(P.*)
FROM UNIVERSIDADE.DEPARTAMENTO DE, UNIVERSIDADE.PROFESSOR P, UNIVERSIDADE.DISCIPLINA D, UNIVERSIDADE.MINISTRA M
WHERE DE.NOME_DEPTO = 'Computação' AND P.SIAPE = M.SIAPE AND M.COD_DISCIPLINA = D.COD_DISCIPLINA AND  D.NOME = 'Fundamentos de Bancos de Dados'; 

--8)
SELECT P.*
FROM  UNIVERSIDADE.MINISTRA M, UNIVERSIDADE.PROFESSOR P
WHERE M.SIAPE = P.SIAPE
GROUP BY P.SIAPE
HAVING COUNT(DISTINCT M.COD_DISCIPLINA) = (
					SELECT COUNT(DISTINCT M.COD_DISCIPLINA)
					FROM UNIVERSIDADE.MINISTRA M);




--9)
SELECT M.COD_DISCIPLINA
FROM  UNIVERSIDADE.MINISTRA M, UNIVERSIDADE.PROFESSOR P
WHERE M.SIAPE = P.SIAPE
GROUP BY M.COD_DISCIPLINA
HAVING  COUNT(DISTINCT M.SIAPE) >1;

--10)
SELECT D.NUMERO_DEPTO, MAX(P.SALARIO)
FROM  UNIVERSIDADE.DEPARTAMENTO D, UNIVERSIDADE.PROFESSOR P
WHERE P.COD_DEPTO = D.NUMERO_DEPTO AND P.CATEGORIA = 'Adjunto I'
GROUP BY D.NUMERO_DEPTO;

--11)

FROM UNIVERSIDADE.PROFESSOR P, UNIVERSIDADE.DISCIPLINA D, UNIVERSIDADE.MINISTRA M
WHERE P.SIAPE = M.SIAPE AND D.COD_DISCIPLINA = M.COD_DISCIPLINA AND D.COD_DISCIPLINA =1
